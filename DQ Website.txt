    <script>
        const todayStr = new Date().toISOString().split("T")[0];
        const prices = { "mini": 21.99, "8inch": 33.99, "10inch": 38.99, "sheet": 46.99 };
        
        const form = document.getElementById('cake-order-form');
        const dateInput = document.getElementById('pickupDate');
        const timeInput = document.getElementById('pickupTime');
        const cakeSize = document.getElementById('cakeSize');
        const tradDesc = document.getElementById('trad-desc');
        const feeRow = document.getElementById('fee-row');
        const totalDisplay = document.getElementById('total-display');

        dateInput.min = todayStr;

        function updateLogic() {
            const type = document.querySelector('input[name="cakeType"]:checked').value;
            const size = cakeSize.value;
            const isSameDay = dateInput.value === todayStr;

            // Toggle Blizzard Flavor dropdown
            document.getElementById('blizzard-flavors').classList.toggle('hidden', type !== 'blizzard');

            // 48h Blizzard Rule
            if (type === 'blizzard' && dateInput.value) {
                const tomorrow = new Date(); tomorrow.setDate(tomorrow.getDate() + 1);
                if (dateInput.value <= tomorrow.toISOString().split("T")[0]) {
                    alert("Blizzard cakes require 48 hours notice. Please call 863-422-9336.");
                    dateInput.value = "";
                }
            }

            // Description Logic
            if (size === 'sheet') {
                tradDesc.textContent = "Feeding 20-24 people, this sheet cake delivers the wow to the party. Layers of world-famous vanilla and chocolate soft serve and signature fudge and crunch center.";
            } else {
                tradDesc.textContent = "Our signature fudge and crunch center surrounded by our world famous vanilla and chocolate soft serve.";
            }

            // Pricing Logic
            const base = prices[size] || 0;
            const fee = (isSameDay && type === 'traditional') ? 5.00 : 0;
            feeRow.classList.toggle('hidden', fee === 0);
            totalDisplay.textContent = base > 0 ? `Total: $${(base + fee).toFixed(2)}` : "Total: $TBD";
        }

        // Event Listeners
        [dateInput, cakeSize].forEach(el => el.addEventListener('change', updateLogic));
        document.querySelectorAll('input[name="cakeType"]').forEach(el => el.addEventListener('change', updateLogic));

        // Submit to Modal
        form.addEventListener('submit', (e) => {
            e.preventDefault();
            const type = document.querySelector('input[name="cakeType"]:checked').value;
            const colors = Array.from(document.querySelectorAll('input[name="borderColor"]:checked')).map(c => c.value).join(", ");
            
            document.getElementById('review-content').innerHTML = `
                <p><strong>Pickup:</strong> ${dateInput.value} at ${timeInput.value}</p>
                <p><strong>Cake:</strong> ${cakeSize.value} ${type}</p>
                <p><strong>Design:</strong> Border: ${colors || 'Standard'}</p>
                <p><strong>Message:</strong> "${document.getElementsByName('cakeMessage')[0].value || 'None'}"</p>
                <hr>
                <p class="text-dqred font-black">${totalDisplay.textContent}</p>
            `;
            document.getElementById('review-modal').classList.remove('hidden');
        });

        // Final Confirm
        document.getElementById('confirm-send').addEventListener('click', async () => {
            const btn = document.getElementById('confirm-send');
            btn.textContent = "SENDING..."; btn.disabled = true;
            
            const response = await fetch(form.action, {
                method: 'POST',
                body: new FormData(form),
                headers: { 'Accept': 'application/json' }
            });

            if (response.ok) {
                document.getElementById('review-modal').classList.add('hidden');
                document.getElementById('form-container').classList.add('hidden');
                document.getElementById('success-message').classList.remove('hidden');
                document.getElementById('success-name').textContent = document.getElementsByName('fullName')[0].value;
                document.getElementById('success-date').textContent = dateInput.value;
            } else {
                alert("Error sending. Please call 863-422-9336.");
                btn.textContent = "Confirm & Send"; btn.disabled = false;
            }
        });
    </script>
